$plainText = "Hello, world!"
$passphrase = "MySecretPassphrase"
$encryptionKey = New-Object System.Security.Cryptography.RijndaelManaged
$encryptionKey.KeySize = 192
$encryptionKey.Mode = [System.Security.Cryptography.CipherMode]::CBC
$encryptionKey.Padding = [System.Security.Cryptography.PaddingMode]::PKCS7
$encryptionKey.GenerateIV()
$encryptionKey.GenerateKey()

# Triple DES Encryption
$tripleDesEncryption = New-Object System.Security.Cryptography.TripleDESCryptoServiceProvider
$tripleDesEncryption.Key = $encryptionKey.Key
$tripleDesEncryption.Mode = [System.Security.Cryptography.CipherMode]::ECB
$tripleDesEncryption.Padding = [System.Security.Cryptography.PaddingMode]::PKCS7

# RC2 Encryption
$rc2Encryption = New-Object System.Security.Cryptography.RC2CryptoServiceProvider
$rc2Encryption.Key = $encryptionKey.Key
$rc2Encryption.Mode = [System.Security.Cryptography.CipherMode]::ECB
$rc2Encryption.Padding = [System.Security.Cryptography.PaddingMode]::PKCS7

# Encrypt with Rijndael
$rijndaelEncryptor = $encryptionKey.CreateEncryptor()
$memoryStream = New-Object System.IO.MemoryStream
$cryptoStream = New-Object System.Security.Cryptography.CryptoStream($memoryStream, $rijndaelEncryptor, [System.Security.Cryptography.CryptoStreamMode]::Write)
$streamWriter = New-Object System.IO.StreamWriter($cryptoStream)
$streamWriter.Write($plainText)
$streamWriter.Flush()
$cryptoStream.FlushFinalBlock()
$memoryStream.Flush()
$rijndaelEncryptedBytes = $memoryStream.ToArray()
$memoryStream.Close()
$cryptoStream.Close()

# Encrypt with Triple DES
$tripleDesEncryptor = $tripleDesEncryption.CreateEncryptor()
$memoryStream = New-Object System.IO.MemoryStream
$cryptoStream = New-Object System.Security.Cryptography.CryptoStream($memoryStream, $tripleDesEncryptor, [System.Security.Cryptography.CryptoStreamMode]::Write)
$cryptoStream.Write($rijndaelEncryptedBytes, 0, $rijndaelEncryptedBytes.Length)
$cryptoStream.FlushFinalBlock()
$tripleDesEncryptedBytes = $memoryStream.ToArray()
$memoryStream.Close()
$cryptoStream.Close()

# Encrypt with RC2
$rc2Encryptor = $rc2Encryption.CreateEncryptor()
$memoryStream = New-Object System.IO.MemoryStream
$cryptoStream = New-Object System.Security.Cryptography.CryptoStream($memoryStream, $rc2Encryptor, [System.Security.Cryptography.CryptoStreamMode]::Write)
$cryptoStream.Write($tripleDesEncryptedBytes, 0, $tripleDesEncryptedBytes.Length)
$cryptoStream.FlushFinalBlock()
$rc2EncryptedBytes = $memoryStream.ToArray()
$memoryStream.Close()
$cryptoStream.Close()

$encryptedText = [Convert]::ToBase64String($rc2EncryptedBytes)
Write-Output "Encrypted text: $encryptedText"
